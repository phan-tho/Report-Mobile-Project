\chapter{Thực nghiệm và đánh giá}
\label{chap:experiment}
\paragraph{Đối với các KLTN giải quyết bài toán nghiên cứu:}

\begin{itemize}
    \item Chương này thường có tiêu đề ``Thực nghiệm và đánh giá''
    \item Các nội dung chính cần trình bày:
    \begin{itemize}
        \item Câu hỏi nghiên cứu
        \item Dữ liệu và thiết lập thực nghiệm
        \item Các phương pháp đối sánh
        \item Độ đo đánh giá
        \item Kết quả thực nghiệm
        \item Phân tích và thảo luận kết quả
    \end{itemize}
\end{itemize}


\paragraph{Đối với các KTLN xây dựng hệ thống:}

\begin{itemize}
    \item Chương này thường có tiêu đề ``Cài đặt, kiểm thử và triển khai hệ thống''
    \begin{itemize}
        \item Các nội dung chính cần trình bày:
        \begin{itemize}
            \item Môi trường triển khai và cài đặt
            \item Kịch bản kiểm thử
            \item Kết quả kiểm thử
            \item Phân tích và thảo luận kết quả
        \end{itemize}
    \end{itemize}
\end{itemize}

\textit{\textbf{Lưu ý:}} Sinh viên cần mô tả phần mình cài đặt, phân tách với phần reuse
