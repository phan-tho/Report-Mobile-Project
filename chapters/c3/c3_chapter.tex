


\chapter{Phương pháp đề xuất}

\paragraph{Đối với các KLTN giải quyết bài toán nghiên cứu:}

\begin{itemize}
    \item Tiêu đề của chương thường là ``Phương pháp đề xuất/Mô hình đề xuất cho bài toán XYZ''. 
    \item Chương này cần trình bày được:
    \begin{itemize}
        \item Tổng quan phương pháp/Kiến trúc tổng thể
        \item Mô hình/thuật toán/phương pháp chi tiết
        \item Cần lý giải được vì sao giải quyết bài toán như vậy, phân tích ưu, nhược điểm 
    \end{itemize}
\end{itemize}

\paragraph{Đối với các KTLN xây dựng hệ thống:}
\begin{itemize}
    \item Tiêu đề của chương thường là ``Phân tích và thiết kế hệ thống XYZ''.
    \item Chương này cần trình bày được:
    \begin{itemize}
        \item Phân tích yêu cầu
        \item Kiến trúc tổng quan của hệ thống
        \item Thiết kế hệ thống chi tiết: Biểu đồ ca sử dụng, các biểu đồ tuần tự, biểu đồ luồng hoạt động, biểu đồ lớp cho các ca sử dụng chính
        \item Thiết kế CSDL
    \end{itemize}
\end{itemize}

Lưu ý:
\begin{itemize}
    \item \textbf{Không} đưa các bảng biểu/hình ảnh không liên quan vào KLTN
    \item Tiêu đề Bảng đặt trên bảng, tiêu đề Hình đặt dưới hình
    \item Toàn bộ hình ảnh, bảng biểu phải được tham chiếu và giải thích rõ ràng. \textbf{Không} chấp nhận các KLTN với các mục chỉ có hình/bảng không có giải thích/mô tả/thảo luận. 
    \item \textbf{Không} viết KLTN theo dạng ``gạch đầu dòng'' để đảm bảo rõ ràng và liền mạch trong việc giải thích và lập luận.
\end{itemize}
